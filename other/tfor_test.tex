		\#\askmaps@opt\#
		\if i\askmaps@opt\def\askmaps@opti{i}\fi%
		\if I\askmaps@opt\def\askmaps@opti{}\fi%
		\if f\askmaps@opt\def\askmaps@optf{f}\fi%
		\if F\askmaps@opt\def\askmaps@optf{}\fi%
		\if b\askmaps@opt\def\askmaps@optb{b}\fi%
		\if B\askmaps@opt\def\askmaps@optb{}\fi%
		\if e\askmaps@opt\def\askmaps@optb{}\def\askmaps@optf{}\def\askmaps@opti{}\fi%
	}%
	%\if i\askmaps@opti Hallo \fi
	%\#\askmaps@opti\#\askmaps@optf\#
}%
